# 1. Внесите правки в проект
# 2. Добавьте изменения в индекс
git add .

# 3. Создайте коммит с описанием
git commit -m "Описание ваших изменений"

# 4. Отправьте на GitHub (теперь без -u)
git push
# ============================================
# 🚀 GIT CHEAT SHEET — PowerShell / Windows
# ============================================

# 🔹 НАСТРОЙКА (один раз)
git config --global user.name "Ваше Имя"           # Имя для коммитов
git config --global user.email "email@example.com" # Email для коммитов
git config --global core.autocrlf true             # Авто-конвертация концов строк (Windows)
git config --global --list                         # Показать все настройки

# 🔹 СОЗДАНИЕ И КЛОНИРОВАНИЕ
git init                                           # Создать новый репозиторий в папке
git clone <URL>                                    # Скопировать удалённый репозиторий
git clone <URL> --branch <name>                    # Клонировать конкретную ветку

# 🔹 ОСНОВНОЙ ЦИКЛ РАБОТЫ
git status                                         # Что изменено? Что в индексе?
git add <файл>                                     # Добавить конкретный файл в индекс
git add .                                          # Добавить ВСЕ изменения в индекс
git add -p                                         # Добавить изменения по частям (интерактивно)
git commit -m "Описание изменений"                 # Создать коммит с сообщением
git commit -am "Описание"                          # Добавить + закоммитить изменённые файлы (кроме новых)
git push                                           # Отправить коммиты на GitHub
git push -u origin <ветка>                         # Отправить и связать локальную ветку с удалённой

# 🔹 ПРОСМОТР ИСТОРИИ
git log                                            # Полная история коммитов
git log --oneline                                  # Кратко: хэш + сообщение
git log --oneline --graph --all                    # История с ветками в виде графика
git log -p                                         # История с показом изменений в файлах
git show <хэш>                                     # Показать детали конкретного коммита

# 🔹 ВЕТКИ (BRANCHES)
git branch                                         # Список локальных веток
git branch -a                                      # Список всех веток (локальных + удалённых)
git branch -m <старое> <новое>                     # Переименовать ветку
git branch -d <ветка>                              # Удалить ветку (безопасно)
git branch -D <ветка>                              # Удалить ветку принудительно

git checkout <ветка>                               # Переключиться на ветку
git checkout -b <новая_ветка>                      # Создать и переключиться на новую ветку
git switch <ветка>                                 # Современная альтернатива checkout (только переключение)
git switch -c <новая_ветка>                        # Создать и переключиться (аналог checkout -b)

git merge <ветка>                                  # Влить изменения из другой ветки в текущую
git merge --no-ff <ветка>                          # Влить с сохранением истории слияния

# 🔹 ОТМЕНА И ИСПРАВЛЕНИЯ
git restore <файл>                                 # Отменить изменения в файле (до git add)
git restore --staged <файл>                        # Убрать файл из индекса (после git add)
git reset --soft HEAD~1                            # Отменить последний коммит, изменения оставить в индексе
git reset --mixed HEAD~1                           # Отменить коммит и индекс, изменения оставить в рабочей папке
git reset --hard HEAD~1                            # ⚠️ Полностью отменить коммит и все изменения (безвозвратно!)
git revert <хэш>                                   # Создать новый коммит, отменяющий изменения указанного

# 🔹 РАБОТА С УДАЛЁННЫМ РЕПОЗИТОРИЕМ (GitHub)
git remote -v                                      # Показать URL удалённых репозиториев
git remote add origin <URL>                        # Добавить удалённый репозиторий с именем "origin"
git remote set-url origin <новый_URL>              # Сменить URL (например, SSH → HTTPS)
git fetch                                          # Скачать изменения с GitHub, но НЕ применять
git pull                                           # Скачать + автоматически влить изменения (fetch + merge)
git pull --rebase                                  # Скачать и применить изменения поверх ваших коммитов

# 🔹 ВРЕМЕННОЕ СОХРАНЕНИЕ ИЗМЕНЕНИЙ (STASH)
git stash                                          # Спрятать изменения во "временное хранилище"
git stash list                                     # Показать список спрятанных изменений
git stash pop                                      # Достать последние изменения и удалить из stash
git stash apply                                    # Достать изменения, но оставить в stash

# 🔹 .GITIGNORE — ЧТО ИГНОРИРОВАТЬ
# Добавьте в файл .gitignore в корне проекта:
#
# bin/ obj/ *.pdb *.user          # Visual Studio / C#
# node_modules/ *.log             # Node.js
# __pycache__/ *.pyc .env         # Python
# .vs/ *.suo *.sln.docstates      # VS служебные
# .DS_Store Thumbs.db             # Системные файлы
# *.tmp *.temp backup/            # Временные файлы

# 🔹 ПОЛЕЗНЫЕ СОВЕТЫ
# • Делайте коммиты часто и с понятными сообщениями
# • Перед началом работы: git pull (чтобы получить чужие правки)
# • Перед пушем: git status (убедиться, что не отправляете лишнее)
# • Для отмены ошибок: git refshow (показывает историю всех действий)

# 🔹 ЕСЛИ ЧТО-ТО ПОШЛО НЕ ТАК
git status                                         # Первый шаг — посмотреть, что происходит
git log --oneline -5                               # Посмотреть последние 5 коммитов
git diff                                           # Показать изменения, которые ещё не в индексе
git diff --staged                                  # Показать изменения, которые уже в индексе
