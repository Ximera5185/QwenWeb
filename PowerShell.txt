или
(Лучший вариант)

$output = "project_context.txt"
$exts = "*.cs","*.csproj","*.razor","*.json","*.config","*.xml","*.md","*.txt","*.yml"
$exclude = "bin|obj|\.git|\.vs|node_modules|wwwroot|Storage|Migrations|project_context.txt|project_dump.txt|monitor_test.db"
$maxSize = 5MB

$files = Get-ChildItem -Recurse -Include $exts | Where-Object { $_.FullName -notmatch $exclude }
$total = $files.Count
$counter = 0

"=== PROJECT CONTEXT DUMP ===" | Out-File -FilePath $output -Encoding UTF8 -Force

foreach ($f in $files) {
    $counter++
    # Пропускаем большие файлы (логи, дампы, кэш)
    if ($f.Length -gt $maxSize) { continue }

    $rel = $f.FullName.Replace((Get-Location).Path, '').TrimStart('\')
    "`n`n### FILE: $rel ###`n" | Out-File -FilePath $output -Append -Encoding UTF8
    Get-Content $f.FullName -Raw -Encoding UTF8 | Out-File -FilePath $output -Append -Encoding UTF8

    # Индикатор прогресса (обновляется в той же строке)
    $pct = [math]::Round(($counter/$total)*100)
    Write-Host "[$counter/$total] $pct% - $rel" -NoNewline
    Write-Host "`r" -NoNewline
}

Write-Host "`n✅ Готово! Файл: $output" -ForegroundColor Green
Write-Host "📦 Размер: $([math]::Round((Get-Item $output).Length/1KB,1)) KB" -ForegroundColor Cyan




1. Get-ChildItem (dir, ls)
   Описание: Показать файлы и папки в текущей директории.
   Пример: dir C:\Projects

2. Set-Location (cd)
   Описание: Перейти в указанную папку.
   Пример: cd Desktop
           cd .. (выйти на уровень выше)

3. New-Item (ni)
   Описание: Создать новый файл или папку.
   Пример: ni readme.txt (создаст файл)
           ni Logs -ItemType Directory (создаст папку)

4. Remove-Item (rm, del)
   Описание: Удалить файл или папку.
   Пример: rm temp.txt
           rm OldFolder -Recurse -Force (удалит папку со всем содержимым)

5. Copy-Item (cp)
   Описание: Скопировать файл или папку.
   Пример: cp report.txt D:\Backup\

6. Move-Item (mv)
   Описание: Переместить или переименовать файл/папку.
   Пример: mv old.txt new.txt (переименование)
           mv script.ps1 C:\Scripts\ (перемещение)

7. Test-Path
   Описание: Проверить, существует ли файл или папка.
   Пример: Test-Path C:\data\config.json
   Результат: True (есть) или False (нет)

8. Get-Content (cat, type)
   Описание: Вывести содержимое текстового файла в консоль.
   Пример: type log.txt
           cat notes.txt

9. Set-Content / Add-Content
   Описание: Записать (перезаписать) или добавить текст в файл.
   Пример: "Привет" | Set-Content hi.txt
           "Вторая строка" | Add-Content hi.txt

10. Get-Process (gps)
    Описание: Показать список запущенных процессов.
    Пример: gps
            gps -Name chrome (только процессы Chrome)

11. Stop-Process (kill)
    Описание: Принудительно завершить зависшую или ненужную программу.
    Пример: kill -Name notepad
            Stop-Process -Id 1234

12. Clear-Host (cls)
    Описание: Очистить экран консоли от предыдущего вывода.
    Пример: cls

13. Get-Help
    Описание: Получить справку по любой команде.
    Пример: help Get-ChildItem
            help rm -Examples (показать только примеры использования)

14. Select-String (sls)
    Описание: Найти текст внутри файлов (аналог grep).
    Пример: sls "ошибка" *.log
            Get-Content .\file.txt | sls "ключевое слово"

15. Get-Command (gcm)
    Описание: Найти команду по названию или части слова.
    Пример: gcm *service* (все команды со словом service)
            gcm -Noun Process (все команды, работающие с процессами)

💡 ПОДСКАЗКИ:
• Нажмите Tab для автодополнения команд и путей.
• Стрелка вверх/вниз листает историю команд.
• Команды не чувствительны к регистру (dir, DIR, Dir работают одинаково).
• Если в пути есть пробелы, обязательно используйте кавычки: "C:\My Folder"
